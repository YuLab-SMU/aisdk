背景

本报告模拟一家中型电商平台 Q4（2024年10月—12月）的销售数据， 展示如何用 {ai} 代码块驱动完整的分析工作流：

  • 数据加载 → AI 生成清洗代码 → AI 解读统计摘要 → AI 绘图 → AI 撰写洞察

每个 {ai} 块共享同一个 session = "sales"，因此 AI 拥有完整对话上下文， 可以像真实分析师一样持续跟进。

2. 数据清洗

AI 任务：检查 orders 的数据质量，生成清洗代码。

Code 
请检查 orders 数据框的数据质量，重点关注：
1. 是否有缺失值
2. revenue 分布是否合理（退货订单应为负值）
3. date 字段是否覆盖完整的 Q4 区间

生成一段完整的 R 清洗代码，产出一个干净的 orders_clean 数据框，
并打印出清洗前后的行数对比。
```r
# Required libraries
library(dplyr)
library(lubridate)

# Start cleaning/checking pipeline
# Assumes `orders` exists in the environment

# 1. Basic counts
rows_before <- nrow(orders)
cat("清洗前行数 (rows before):", rows_before, "\n\n")

# 2. 缺失值检查（包括空字符串）
na_counts <- sapply(orders, function(col) {
  if (is.character(col)) {
    sum(is.na(col) | trimws(col) == "")
  } else {
    sum(is.na(col))
  }
})
cat("每列缺失/空值计数:\n")
print(na_counts)
cat("\n")

# 3. 确保 date 是 Date 类型，尝试多种格式解析（保留原始 NA）
orders <- orders |>
  mutate(
    date = case_when(
      inherits(date, "Date") ~ date,
      TRUE ~ {
        # 先尝试常见 ISO 格式 ymd，再 fallback 到通用解析
        parsed <- suppressWarnings(ymd(as.character(date)))
        parsed[is.na(parsed)] <- as.Date(suppressWarnings(parse_date_time(as.character(date),
                                                                          orders = c("Y-m-d", "Y/m/d", "m/d/Y", "d/m/Y", "Ymd", "ymd HMS")))[is.na(parsed)])
        parsed
      }
    )
  )

# 报告 date 范围
date_min <- min(orders$date, na.rm = TRUE)
date_max <- max(orders$date, na.rm = TRUE)
cat("date 范围: ", as.character(date_min), " 至 ", as.character(date_max), "\n\n")

# 4. 检查每个年份的 Q4 (10-01 至 12-31) 覆盖情况（按年份）
years_present <- sort(unique(year(orders$date[!is.na(orders$date)])))
if (length(years_present) == 0) {
  cat("警告: 数据中没有有效的 date 值，可以跳出后续 Q4 覆盖检查。\n\n")
} else {
  cat("按年份检查 Q4(10-01 ~ 12-31) 覆盖情况:\n")
  for (yr in years_present) {
    q4_seq <- seq(as.Date(paste0(yr, "-10-01")), as.Date(paste0(yr, "-12-31")), by = "day")
    present_dates <- unique(orders$date[year(orders$date) == yr])
    if (length(present_dates) == 0) {
      coverage_pct <- 0
      missing_dates <- q4_seq
    } else {
      missing_dates <- setdiff(q4_seq, present_dates)
      coverage_pct <- 100 * (1 - length(missing_dates) / length(q4_seq))
    }
    cat(sprintf(" %d 年: Q4 天数 = %d, 覆盖天数 = %d, 覆盖率 = %.1f%%\n",
                yr, length(q4_seq), length(q4_seq) - length(missing_dates), coverage_pct))
    if (length(missing_dates) > 0) {
      cat("   缺失的 Q4 日期示例 (最多显示 5): ", paste(head(as.character(missing_dates), 5), collapse = ", "), "\n")
    }
  }
  cat("\n")
}

# 5. revenue 分布与退货标志一致性检查
cat("revenue 基本统计（整个数据集）:\n")
print(summary(orders$revenue))
cat("\n按 is_return 分组的 revenue 统计:\n")
revenue_by_return <- orders |>
  mutate(is_return = ifelse(is.na(is_return), NA, as.logical(is_return))) |>
  group_by(is_return) |>
  summarise(
    n = n(),
    min = min(revenue, na.rm = TRUE),
    q1 = quantile(revenue, 0.25, na.rm = TRUE),
    median = median(revenue, na.rm = TRUE),
    mean = mean(revenue, na.rm = TRUE),
    q3 = quantile(revenue, 0.75, na.rm = TRUE),
    max = max(revenue, na.rm = TRUE),
    .groups = "drop"
  )
print(revenue_by_return)
cat("\n")

# 找出不符合约定的行（退货应为负，非退货应为正或零）
bad_returns <- orders |>
  filter(!is.na(is_return) & is_return == TRUE & !is.na(revenue) & revenue >= 0)
bad_sales <- orders |>
  filter(!is.na(is_return) & is_return == FALSE & !is.na(revenue) & revenue < 0)
cat("退货标记为 TRUE 但 revenue >= 0 的行数:", nrow(bad_returns), "\n")
if (nrow(bad_returns) > 0) {
  cat("示例 (最多 5 行):\n")
  print(head(bad_returns, 5))
}
cat("退货标记为 FALSE 但 revenue < 0 的行数:", nrow(bad_sales), "\n")
if (nrow(bad_sales) > 0) {
  cat("示例 (最多 5 行):\n")
  print(head(bad_sales, 5))
}
cat("\n")

# 6. 开始清洗：
#  - 修正 revenue 符号： is_return == TRUE -> revenue = -abs(revenue)
#                       is_return == FALSE -> revenue = abs(revenue)
#  - units 取绝对值（负数单位很可能是输入错误）
#  - 删除关键字段缺失的行（order_id, date, revenue, units, is_return）
#  - 删除重复 order_id（保留第一条）
orders_fixed <- orders |>
  mutate(
    # 修正 revenue 符号（保留 NA）
    revenue = case_when(
      is.na(revenue) ~ NA_real_,
      is.na(is_return) ~ revenue, # 如果 is_return 本身缺失，先不改 revenue (后面会被视为关键缺失并删除)
      is_return ~ -abs(as.numeric(revenue)),
      !is_return ~ abs(as.numeric(revenue))
    ),
    # units 取绝对值（保留 NA）
    units = ifelse(is.na(units), NA_integer_, as.integer(abs(units)))
  )

# 记录因关键字段缺失被删除的数量
before_drop_missing <- nrow(orders_fixed)
orders_nonmissing <- orders_fixed |>
  filter(
    !is.na(order_id) & trimws(as.character(order_id)) != "",
    !is.na(date),
    !is.na(revenue),
    !is.na(units),
    !is.na(is_return)
  )
dropped_missing <- before_drop_missing - nrow(orders_nonmissing)

# 删除重复 order_id（保留出现顺序的第一条）
orders_nodup <- orders_nonmissing |>
  arrange(order_id, date) |>
  distinct(order_id, .keep_all = TRUE)
dups_removed <- nrow(orders_nonmissing) - nrow(orders_nodup)

# 最终的清洁数据框
orders_clean <- orders_nodup

# 7. 输出清洗结果摘要
rows_after <- nrow(orders_clean)
cat("清洗后行数 (rows after):", rows_after, "\n")
cat("共删除行数:", rows_before - rows_after, "(其中因缺失关键字段删除:", dropped_missing,
    ", 因重复 order_id 删除:", dups_removed, ")\n\n")

cat("orders_clean 变量已创建，可直接使用。\n")
cat("清洗后 revenue 按 is_return 的简单统计（供核验）:\n")
print(
  orders_clean |>
    group_by(is_return) |>
    summarise(n = n(), min = min(revenue), median = median(revenue), mean = mean(revenue), max = max(revenue), .groups = "drop")
)

# 显示前几行作为检查
cat("\norders_clean 示例 (前 6 行):\n")
print(head(orders_clean, 6))
```

4.2 品类 GMV 占比

AI 任务：用 by_cat 画饼图或环形图。

Code 
请用 by_cat 数据绘制一张品类 GMV 占比的环形图（donut chart）：
- 用 ggplot2 实现，geom_col + coord_polar
- 显示每个品类的名称和百分比
- 使用 RColorBrewer 的 Set2 配色
- 无图例，在扇区内直接标注文字
- 标题"Q4 品类 GMV 占比"
```r
# 绘图所需库
library(ggplot2)
library(dplyr)
library(scales)
library(RColorBrewer)

# 确保数据存在
if (!exists("by_cat")) stop("对象 'by_cat' 不存在，请先加载数据框 by_cat。")

# 准备数据：按 GMV 降序，计算占比与标签
by_cat_plot <- by_cat |>
  arrange(desc(gmv)) |>
  mutate(
    category = factor(category, levels = unique(category)),
    pct = gmv / sum(gmv),
    label = paste0(category, "\n", percent(pct, accuracy = 0.1))
  )

# 绘制环形图（donut）
p <- ggplot(by_cat_plot, aes(x = 2, y = gmv, fill = category)) +
  geom_col(width = 1, color = "white") +
  coord_polar(theta = "y") +
  xlim(0.5, 2.5) +                                   # 留出中间空白形成环形
  geom_text(aes(label = label),                      # 在扇区内标注品类与百分比
            position = position_stack(vjust = 0.5),
            size = 4) +
  scale_fill_brewer(palette = "Set2") +
  theme_void() +
  theme(
    legend.position = "none",
    plot.title = element_text(hjust = 0.5, face = "bold")
  ) +
  ggtitle("Q4 品类 GMV 占比")

# 显示图形
print(p)
```

附：分析工作流说明

本报告共发起 16 轮 AI 对话，所有上下文在 session='sales' 中共享传递。
Note

关键设计模式

  1. 数据先行，AI 后解读：先用 {r} 块计算出干净的汇总表（monthly、by_cat 等），再让 {ai} 块引用这些变量——引擎自动检测变量名并传入结构摘要，AI 就能写出精准代码。

  2. 单一 session 贯穿全文：所有 {ai} 块共享 session="sales"，AI 记得前面做了什么，不需要反复重交代背景。

  3. echo/eval 灵活控制：

    • 纯解读文字：echo=false（不显示 prompt）
    • 生成可复用代码：echo=true, eval=true（展示代码 + 运行结果）

  4. 人机协作：人负责数据管道（加载、清洗骨架、维度设计），AI 负责代码生成和文字解读，各司其职。
